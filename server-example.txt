// Backend server example (Node.js) - place in separate repo or server
import express from 'express';
import cors from 'cors';
import aiRoute from './ai.js';
const app = express();
app.use(cors());
app.use(express.json());
app.use('/api/ai', aiRoute);
app.listen(process.env.PORT||5000, ()=>console.log('AI server running'));
